\appendix{Model Pseudocode and Workflow}
\begin{algorithm}[htbp]
\caption{Fabric Consumption Forecasting Training Pipeline (12 steps)}
\begin{algorithmic}[1]
\State Load CSV: 5{,}000 rows $\times$ 34 columns
\State Encode categoricals (alphabetical LabelEncoder); dropna on raw columns
\State Extract $X$ (9 features), $y$ (Actual\_Consumption\_yards)
\State Split 64/16/20, \texttt{random\_state=42}
\State Fit StandardScaler on $X_{\text{train}}$; transform $X_{\text{val}}, X_{\text{test}}$
\State Train LR $\to$ RF (300 trees, depth 12) $\to$ XGBoost (300 trees, $\eta$=0.08, depth 8) $\to$ LSTM (200 eps, pat.=15)
\State Evaluate: RMSE, MAE, R\textsuperscript{2}, MAPE on test set
\State Build ensemble: $w_k = 1/\text{RMSE}_k^2 \div \sum_j 1/\text{RMSE}_j^2$ (dynamic inverse-RMSE\textsuperscript{2})
\State Run 5-fold CV on train+validation partition (4{,}000 orders) for LR, RF, XGBoost
\State Extract feature importances from XGBoost, Random Forest
\State Save models (.pkl/.h5), scaler.pkl, encoders.pkl, model\_metadata.json
\State Verify all 8 output files present and non-zero
\end{algorithmic}
\end{algorithm}
